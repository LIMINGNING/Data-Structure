#define MaxSize 1000
typedef int ElementType;  /* 默认元素可以 用非负整数表示 */

void Initialization(S,n)

void Input_connection(SetType S)
{
    ElementType u,v;
    SetName Root1,Root2;
    scanf("%d %d\n",&u,&v);
    Root1=Find(S,u-1);
    Root2=Find(S,v-1);
    if(Root1!=Root2)
        Union(S,Root1,Root2);
}

{
    ElementType u,v;
    SetName Root1,Root2;
    scanf("%d %d\n",&u,&v);
    Root1=Find(S,u-1);
    Root2=Find(S,v-1);
    if(Root1==Root2)
        printf("yes\n");
    else
        printf("no\n");
}

    int i,counter=0;

    {

            counter++;
    }
    if(counter==1)
